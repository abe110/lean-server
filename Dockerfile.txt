FROM ubuntu:22.04

# Install system dependencies
RUN apt-get update && apt-get install -y \
    curl \
    git \
    build-essential \
    nodejs \
    npm \
    coreutils \
    && rm -rf /var/lib/apt/lists/*

# Install LEAN 4
RUN curl -sSf https://raw.githubusercontent.com/leanprover/elan/master/elan-init.sh | sh -s -- -y
ENV PATH="/root/.elan/bin:${PATH}"
RUN /root/.elan/bin/elan install leanprover/lean4:stable
RUN /root/.elan/bin/elan default leanprover/lean4:stable

# Verify LEAN installation
RUN lean --version

# Set up Node.js app
WORKDIR /app
COPY package*.json ./
RUN npm install

# Copy server code
COPY server.js ./

# Create temp directory
RUN mkdir -p /tmp

# Expose port
EXPOSE 3000

# Start server
CMD ["npm", "start"]
